Example literate config, used by ~$nix build .#doom-emacs-tangle~.

;#+begin_src emacs-lisp
;(doom! :lang org)
;#+end_src

#+begin_src emacs-lisp :tangle config.el
(setq org-latex-src-block-backend 'listings))
#+end_src

tabsize 4:
#+begin_src emacs-lisp :tangle config.el
(setq-default indent-tabs-mode nil)
(setq-default tab-width 4)
(setq indent-line-function 'insert-tab)
#+end_src

Setting transparency:
#+begin_src emacs-lisp :tangle config.el
(set-frame-parameter (selected-frame) 'alpha-background 88)
(add-to-list 'default-frame-alist '(alpha-background . 88))

(set-frame-parameter (selected-frame) 'alpha 88)
(add-to-list 'default-frame-alist '(alpha 88))
#+end_src

Setting alternative Splash image:
#+begin_src emacs-lisp :tangle config.el
(setq fancy-splash-image (concat doom-private-dir "splash/doomEmacsGruvbox.svg"))
#+end_src

Hooks and settings for additional modes:

#+begin_src emacs-lisp :tangle config.el
(use-package! org-modern
  :config
  (add-hook 'org-mode-hook 'org-modern-mode))
  (add-hook 'org-mode-hook 'org-indent-mode))
#+end_src

#+begin_src emacs-lisp :tangle config.el
